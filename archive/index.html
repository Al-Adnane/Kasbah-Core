Direct API wrapper

async function safeSQLQuery(sql) {
  // Ask permission first
  const { ticket, allowed } = 
    await kasbah.decide({
      tool: 'db.query',
      signals: await getSafetySignals()
    });
  
  if (!allowed) throw new Error('Blocked');
  
  // Execute with proof
  return db.query(sql, { 
    kasbah_ticket: ticket 
  });
}
